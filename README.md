def bubble_sort(arr):
    """
    Сортировка пузырьком с оптимизацией: если на проходе не было ни одной
    перестановки, значит массив уже отсортирован, и алгоритм завершает работу.

    Параметры:
    arr (list): список чисел для сортировки

    Возврат:
    None (сортировка выполняется "на месте")
    """
    n = len(arr)
    for i in range(n):
        swapped = False  # флаг, отслеживающий, были ли перестановки на этом проходе
        # проходим по неотсортированной части массива
        for j in range(0, n - i - 1):
            if arr[j] > arr[j + 1]:
                # если элементы в неправильном порядке, меняем их местами
                arr[j], arr[j + 1] = arr[j + 1], arr[j]
                swapped = True
        # если на этом проходе перестановок не было, массив уже отсортирован
        if not swapped:
            break
